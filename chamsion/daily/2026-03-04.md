```cpp
// 触发AI控制（打开/关闭AI）
// ai_cmd: 0x01=打开AI，0x00=关闭AI
void trigger_ai_control(u8 ai_cmd) {
    // 1. 构造符合协议格式的AI控制数据包（和蓝牙SPP接收的格式一致）
    u8 ai_packet[9] = {
        0xAA,        // PROTOCOL_SOF
        0x02,        // PROTOCOL_ID_APP
        0x03,        // PROTOCOL_ID_BOARD
        0x01,        // CMD_AI_CONTROL
        0x00, 0x01,  // 数据长度（1字节）
        ai_cmd,      // AI控制指令（0x01打开/0x00关闭）
        0x00, 0x00   // CRC占位（调试可忽略，因为直接投递消息）
    };
    
    // 2. 填充全局缓存（代码中处理逻辑依赖这些缓存）
    memcpy(protocol_packet_cache, ai_packet, sizeof(ai_packet));
    protocol_packet_cache_len = sizeof(ai_packet);
    
    // 3. 投递消息触发USER_PROTOCOL_AI_CONTROL分支
    // 参数说明：
    // 参数1：目标任务名（固定）
    // 参数2：优先级（固定为4）
    // 参数3：数据包长度（传9，和ai_packet长度一致）
    // 参数4：消息类型（USER_PROTOCOL_AI_CONTROL）
    // 参数5：指令码（CMD_AI_CONTROL=0x01）
    // 参数6：数据长度（1字节）
    os_taskq_post_msg(USER_SPP_TASK_NAME, 4, 9, USER_PROTOCOL_AI_CONTROL, 0x01, 1);
    
    printf("已触发AI控制，指令：%sAI\r\n", ai_cmd == 0x01 ? "打开" : "关闭");
```
