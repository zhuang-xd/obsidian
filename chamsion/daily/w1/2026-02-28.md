```cpp
void get_file_name(char *buffer, uint8_t length, Caller filename)
{
  struct timespec ts;
  clock_gettime(CLOCK_REALTIME, &ts);
  uint64_t now_sec = ts.tv_sec;
  char time_str[64] = {0};
  time_t time_val = (time_t)now_sec;
  struct tm* timeinfo = localtime(&time_val);
  if (timeinfo != nullptr) {
      strftime(time_str, sizeof(time_str), "%Y-%m-%d-%H-%M-%S", timeinfo);
      LOGD("Generated file name time: %s", time_str);
  }
  if (filename == RECORD) {
    snprintf(buffer, length, "%s.mp4", time_str);
  } else if (filename == PHOTO) {
    snprintf(buffer, length, "%s.jpg", time_str);
  }
}

void tws_sync_bat_level(void)
{
#if (TCFG_USER_TWS_ENABLE && BT_SUPPORT_DISPLAY_BAT)
    // 发送两个字节：本机百分比和对端百分比（bit7充电标志）
    u8 local_percent = cur_battery_level;
    u8 percent_level = local_percent;

    if (get_charge_online_flag()) {
        percent_level |= BIT(7);
    }

    u8 data[2];
    data[0] = local_percent;
    data[1] = percent_level;
    tws_api_send_data_to_sibling(data, 2, TWS_FUNC_ID_VBAT_SYNC);

    log_info("tws_sync_bat_level: %d,%d\n", local_percent, percent_level);

}


// 将当前电量转换为0~100%发送给手机同步电量
//（以前版本只返回10的倍数，本次改为精确值以便显示例如87%，86%等）
// bat_val 参数仍然传入，但直接使用 get_vbat_percent() 计算精度百分比
u8  battery_value_to_phone_level(u16 bat_val)
{
    (void)bat_val;
    u8 vbat_percent = get_vbat_percent();
    // clamp just in case
    if (vbat_percent > 100) {
        vbat_percent = 100;
    }
    return vbat_percent;
}


// 返回当前电量百分比(0~100)，包括TWS同步逻辑
// 以前返回的是1~9的等级, 外部根据此值乘10再加10得到 10/20/.../100
// 导致手机端显示只能以10为步长。本函数改为直接返回百分比
u8  get_cur_battery_level(void)
{
    u8 self_percent = get_vbat_percent();
#if TCFG_USER_TWS_ENABLE
    u8 sib_percent = tws_sibling_bat_percent_level;

    /* sibling未同步时值为0xff, 此时直接返回本机电量 */
    if (sib_percent == 0xff) {
        return self_percent;
    }

#if (CONFIG_DISPLAY_TWS_BAT_TYPE == CONFIG_DISPLAY_TWS_BAT_LOWER)
    return (self_percent < sib_percent) ? self_percent : sib_percent;
#elif (CONFIG_DISPLAY_TWS_BAT_TYPE == CONFIG_DISPLAY_TWS_BAT_HIGHER)
    return (self_percent < sib_percent) ? sib_percent : self_percent;
#elif (CONFIG_DISPLAY_TWS_BAT_TYPE == CONFIG_DISPLAY_TWS_BAT_LEFT)
    return (tws_api_get_local_channel() == 'L') ? self_percent : sib_percent;
#elif (CONFIG_DISPLAY_TWS_BAT_TYPE == CONFIG_DISPLAY_TWS_BAT_RIGHT)
    return (tws_api_get_local_channel() == 'R') ? self_percent : sib_percent;
#else
    return self_percent;
#endif // CONFIG_DISPLAY_TWS_BAT_TYPE

#else  //TCFG_USER_TWS_ENABLE == 0
    return self_percent;
#endif
}
```
